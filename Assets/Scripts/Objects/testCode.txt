float speed = 100;
int b = 0;

function rotate(sped) {
    rotation = rotation + sped * deltaTime();
    return rotation;
}

function main() {
    while(b < 2) {
        if (rotation < 360) {    
            print(rotate(speed));
        } else {
            b = b + 1;
            rotation = 0;
        }
    }
}
