float speed = 100;
bool b = true;

function rotate(sped) {
    rotation = rotation + sped * deltaTime();
    return rotation;
}


while (rotation < 360 && b) {
    if (b) {    
        print(rotate(speed));
    } else {
        if (b) {
            b = false;
            rotation = 0;
        }
    }
}
