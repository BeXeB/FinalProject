float speed = 100;

function Setup()
{
    SetTarget(GetIntruderID());
    //SetTarget(GetIDSID());
}

function Main() {
    Rotate(speed);
    if(rotation > 360){
        rotation = 0;
    }
}

function Rotate(sped) {
    rotation = rotation + sped * DeltaTime();
    return rotation;
}
