float speed = 100;
float a;

void Setup()
{
    SetTarget(GetIntruderID());
    Print(a);
}

void Main() {
    SetActive(true);
    Rotate(speed);
    if(rotation > 360){
        rotation = 0;
    }
}

float Rotate(float sped) {
    rotation = rotation + sped * DeltaTime();
    return rotation;
}
